doSomething().then( () => {

### 错误处理

`Promise` 会自动捕获内部异常，并交给 `rejected` 响应函数处理

错误处理的两种方式

1. `reject('错误信息').then(null, msg => {})`
2. `throw new Error('错误信息').catch( msg => {})`

```js
new Promise( resolve => {
    setTimeout( () => {
        throw new Error('bye');
    }, 2000);
})
    .then( value => console.log(`${value} wolrd`))
    .catch( error => console.log(`Error: ${error.message}`));

new Promise( (resolve, reject) => {
    setTimeout( () => {
        reject('bye');
    }, 2000);
})
    .then( null, (msg) => console.log(`${msg} `))
```

在 `catch()` 之后接 `then()` 什么结果

```js
new Promise( (resolve, reject) => {
    setTimeout( () => {
        // resolve();
        reject('reject');
    }, 2000);
})
    .then( (value) => {
        console.log(`${value}`);
        throw new Error('test error');
    })
    .catch( err => {
        console.log(`catch 1: ${err}`);
        throw new Error('test error 2');
    })
    .then( () => {
        console.log('then aftre catch')
    })
    .catch( err => {
        console.log(`catch 2: ${err}`)
    })
```

### `Promise` 常用函数

- `Promise.all`

```js
// 实现队列

function doThing(thing, callback) {
    console.log(thing);
    callback();
}

function queue(things) {
    let promise = Promise.resolve();
    things.forEach( thing => {
        promise = promise.then( () => {
            return new Promise( resolve => {
                doThing(thing, () => {
                    resolve();
                })
            })
        })
    });
    return promise;
}

function queue(things) {
    return things.reduce( (promise, thing) => {
        return promise.then( () => {
            return new Promise( resolve => {
                doThing(thing, () => {
                    resolve();
                });
            });
        });
    }, Promise.resolve());
}
```
