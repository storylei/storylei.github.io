### 其他

- flatten

```js
function flatten(arr) {
    return _flatten(arr, []);
    function _flatten(arr, flat) {
        for (let e of arr) {
            if (Array.isArray(e)) {
                _flatten(e, flat);
            } else {
                flat.push(e);
            }
        }
        return flat;
    }
}
flatten([1, [2, 3], [4, [5, 6]]]);
```
